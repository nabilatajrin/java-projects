public class StringToUpperCase {
    public static void main(String[] args){
        String strUcase = "Hello World".toUpperCase();
        String strLcase = "HelloWorld".toLowerCase();
        
        System.out.println(strUcase); // HELLO WORLD
        System.out.println(strLcase); // helloworld
    }
}
